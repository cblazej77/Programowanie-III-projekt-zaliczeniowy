Srednia dla wszystkich uczniow z matmy
SELECT uz.login, uz.imie, AVG(o.ocena), p.nazwa FROM oceny o INNER JOIN przedmioty p ON o.idp = p.idp INNER JOIN uczniowie u ON u.iducz=o.idu
INNER JOIN uzytkownicy uz ON uz.idu = u.iduucznia WHERE p.nazwa = 'matematyka' GROUP BY uz.login, uz.imie, p.nazwa ORDER BY uz.login;
